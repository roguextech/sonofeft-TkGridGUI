380     def handle_scroll_logic(self):
381         
382         must_rebuild = False
383         
384         if self.cobj.user_tkOptionD.get('scrolly','no') == 'yes':
385             self.maybe_add_y_scroll()
386         elif self.has_y_scroll:
387             must_rebuild = True # request is "no", but already have y scroll
388             
389         if self.cobj.user_tkOptionD.get('scrollx','no') == 'yes':
390             self.maybe_add_x_scroll()
391         elif self.has_x_scroll:
392             must_rebuild = True # request is "no", but already have x scroll
393         
394         if must_rebuild:
395             self.destroy_children()
396             self.put_native_widget_into_pw_frame()
